\section{Message Types}\label{sec:message-types}
The following message types are defined in the protocol:

\begin{center}
\begin{longtable}{c l l l  p{6cm}}
\toprule
\textbf{Enum} & \textbf{Message Type} & \textbf{Data Payload} & \textbf{Direction}&\textbf{Description} \\
\endhead
 
\multicolumn{5}{l}{\textit{Node Discovery and Information}} \\
0 & \texttt{node\_info\_req}\label{msg:node-info-req} & No payload & Server $\rightarrow$ Node & Request node information \\
1 & \texttt{node\_info\_announcement}\label{msg:node-info-announcement} & NodeInfoRes (\ref{struct:NodeInfoRes}) & Node $\rightarrow$ Server & Response with node capabilities and identification \\
\midrule
\multicolumn{5}{l}{\textit{Status Messages}} \\
10 & \texttt{info\_status}\label{msg:info-status} & Status (\ref{struct:Status}) & Node $\rightarrow$ Server & Informational status message \\
11 & \texttt{warning\_status}\label{msg:warning-status} & Status (\ref{struct:Status}) & Node $\rightarrow$ Server & Warning status message \\
12 & \texttt{error\_status}\label{msg:error-status} & Status (\ref{struct:Status}) & Node $\rightarrow$ Server & Error status message \\
\midrule
\multicolumn{5}{l}{\textit{Field Registration}} \\
20 & \texttt{telemetryValue\_registration}\label{msg:telemetryValue-registration} & FieldRegistration (\ref{struct:FieldRegistration}) & Node $\rightarrow$ Server & Register a telemetryValue field \\
21 & \texttt{parameter\_registration}\label{msg:parameter-registration} & FieldRegistration (\ref{struct:FieldRegistration}) & Node $\rightarrow$ Server & Register a parameter field \\
\midrule
\multicolumn{5}{l}{\textit{Telemetry Group Management}} \\
30 & \texttt{telemetry\_group\_definition}\label{msg:parameter-group-definition} & TelemetryGroupDefinition (\ref{struct:TelemetryGroupDefinition}) & Node $\rightarrow$ Server & Define a group of fields for batch updates \\
31 & \texttt{telemetry\_group\_update}\label{msg:parameter-group-update} & TelemetryGroupUpdate (\ref{struct:TelemetryGroupUpdate}) & Node $\rightarrow$ Server & Update values for a field group \\
\midrule
\multicolumn{5}{l}{\textit{Parameter Management}} \\
50 & \texttt{parameter\_set\_req}\label{msg:parameter-set-req} & ParameterSetReq (\ref{struct:ParameterSetReq}) & Server/Node $\rightarrow$ Node & Request to set a parameter value \\
51 & \texttt{parameter\_set\_confirmation}\label{msg:parameter-set-confirmation} & ParameterSetConfirmation (\ref{struct:ParameterSetConfirmation}) & Node $\rightarrow$ Server/Node & Response with confirmed parameter value \\
52 & \texttt{parameter\_set\_lock\_req}\label{msg:parameter-set-lock-req} & ParameterSetLock (\ref{struct:ParameterSetLock}) & Server/Node $\rightarrow$ Node & Request to lock a parameter \\
53 & \texttt{parameter\_set\_lock\_confirmation}\label{msg:parameter-set-lock-res} & ParameterSetLock (\ref{struct:ParameterSetLock}) & Node $\rightarrow$ Server & Response confirming parameter lock \\
\midrule
\multicolumn{5}{l}{\textit{Field Access}} \\
60 & \texttt{field\_get\_req}\label{msg:field-get-req} & FieldGetReq (\ref{struct:FieldGetReq}) & Server/Node $\rightarrow$ Node & Request field value \\
61 & \texttt{field\_get\_res}\label{msg:field-get-res} & FieldGetRes (\ref{struct:FieldGetRes}) & Node $\rightarrow$ Server/Node & Response with field value \\
62 & \texttt{field\_id\_lookup\_req}\label{msg:field-id-lookup-req} & FieldIDLookupReq (\ref{struct:FieldIDLookupReq}) & Node $\rightarrow$ Node & Requests the fieldID of a field name \\
63 & \texttt{field\_id\_lookup\_res}\label{msg:field-id-lookup-res} & FieldIDLookupRes (\ref{struct:FieldIDLookupRes}) & Node $\rightarrow$ Node & Response with field ID \\
\midrule
\multicolumn{5}{l}{\textit{Heartbeat}} \\
40 & \texttt{heartbeat\_req}\label{msg:heartbeat-req} & HeartBeat (\ref{struct:HeartBeat}) & Server $\rightarrow$ Node & Heartbeat request \\
41 & \texttt{heartbeat\_res}\label{msg:heartbeat-res} & HeartBeat (\ref{struct:HeartBeat}) & Node $\rightarrow$ Server & Heartbeat response \\
\midrule
\bottomrule
\end{longtable}
\end{center}
\end{landscape}
